95template <typename T>
96TypedField<T> *FieldManager::registerField(const std::string &name, int dim) {
97 // 幂等：如果同名场已存在，直接返回已有指针
98 auto it = fields_.find(name);
99 if (it != fields_.end()) {
100 return dynamic_cast<TypedField<T> *>(it->second.get());
101 }
102 // 创建新的 TypedField<T> 并注册
103 auto field = std::make_unique<TypedField<T>>(name, dim);
104 TypedField<T> *ptr = field.get();
105 fields_[name] = std::move(field);
106 return ptr;
107}

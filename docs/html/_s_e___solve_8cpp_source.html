18 auto &ctx = this->pdContext_;

22 // 1. 获取仿真参数 (从 YAML)
23 // =================================================================
24 std::string yamlPath = "../../Input/PD.yaml";

28 double totalTime = 1.0;

33 dt = config["Solver"]["TimeStep_dt"].as<double>();
34 if (config["Solver"]["TotalTime"])
35 totalTime = config["Solver"]["TotalTime"].as<double>();
36 if (config["Solver"]["OutputInterval"])
37 outputInterval = config["Solver"]["OutputInterval"].as<int>();
38 }
39
40 int totalSteps = static_cast<int>(totalTime / dt);
41
42 // =================================================================
43 // 2. 获取物理场指针（从 FieldManager）
44 // =================================================================
45 auto &fieldManager = ctx.getFieldManager();
46
47 if (!fieldManager.hasField("Temperature")) {
48 LOG_ERROR("No Temperature field found! Aborting solver.");
49 return;
50 }

53 auto *rateField = fieldManager.getFieldAs<double>("TempRate");
54 auto *fluxField = fieldManager.getFieldAs<double>("HeatFlux");

57 auto &manager = ctx.getParticleManager();
58 auto &bcManager = ctx.getBCManager();
59
60 size_t numParticles = manager.getTotalParticles();
61 double horizon = neighborList.getHorizon();
62
63 // 3D 微导率系数 (Bobaru model): kappa = 6k / (pi * delta^4)
64 // 注意：每个点的材料可能不同，需要在循环内获取 k
65 double PI = 3.14159265358979323846;
66 double horizon4 = std::pow(horizon, 4.0);
67
68 // 获取数据指针 (SoA)
69 const double *volumes = manager.getGeomDataPtr(GRPD::Model::ModelVar::Volume);
70 const double *coords = manager.getGeomDataPtr(GRPD::Model::ModelVar::Coords);
71
72 // 获取 Field 的裸指针（用于 OpenMP 高性能循环）
73 double *tempPtr = tempField->dataPtr();
74 double *ratePtr = rateField->dataPtr();
75 double *fluxPtr = fluxField->dataPtr();
76
77 LOG_INFO("Time loop: Dt = " + std::to_string(dt) +
78 ", TotalSteps = " + std::to_string(totalSteps));
79

81 // 3. 时间步循环
82 // =================================================================
83 for (int step = 0; step <= totalSteps; ++step) {
84
85 // (A) 输出结果

88 this->OutputPD();
89 }

92
93 // (B) 清零变化率
94 rateField->clearToZero();
95 fluxField->clearToZero();
96
97 // (C) 施加所有边界条件中的源项 (含热流、对流)
98 // 源项会累加至 TempRate 和 HeatFlux
99 bcManager.applySources();
100
101 // (C2) 施加强制约束 (Dirichlet) - 第一次施加用于更新初值
102 bcManager.applyConstraints();
103
104// (D) 计算 PD 键相互作用 (并行的热传导内核)
105#pragma omp parallel for schedule(dynamic, 64)
106 for (int i = 0; i < static_cast<int>(numParticles); ++i) {
107 const auto &neighbors = neighborList.getNeighbors(i);
108 if (neighbors.empty())
109 continue;
110
111 // 获取当前粒子材料属性
112 auto *mat = dynamic_cast<GRPD::Material::ThermalMat *>(
113 manager.getParticle(i).getMaterial());
114 if (!mat)
115 continue;
116
117 double k = mat->getConductivity();
118 double kappa_base = (6.0 * k) / (PI * horizon4);
119
120 double Ti = tempPtr[i];
121 const double *pos_i = &coords[3 * i];
122
123 double totalContribution = 0.0;
124
125 for (int j : neighbors) {
126 double Tj = tempPtr[j];
127 const double *pos_j = &coords[3 * j];
128
129 double dx = pos_j[0] - pos_i[0];
130 double dy = pos_j[1] - pos_i[1];
131 double dz = pos_j[2] - pos_i[2];
132 double dist = std::sqrt(dx * dx + dy * dy + dz * dz);
133
134 if (dist < 1e-12)
135 continue; // 避免自耦合或重合点
136
137 // PD 传导公式：kappa * (Tj - Ti) / dist * Vj
138 // 注意：有些公式是 / dist^2，这里按 Bobaru (2012)
139 // 积分项为 \int kappa * (Tj - Ti)/dist dVj
140 double bondContribution = kappa_base * ((Tj - Ti) / dist) * volumes[j];
141 totalContribution += bondContribution;
142 }
143
144 rateField->add(i, totalContribution);
145 fluxField->add(i, totalContribution);
146 }
147
148// (E) 显式时间积分：T_new = T_old + (Rate / (rho * Cp)) * dt
149#pragma omp parallel for
150 for (int i = 0; i < static_cast<int>(numParticles); ++i) {
151 auto *mat = dynamic_cast<GRPD::Material::ThermalMat *>(
152 manager.getParticle(i).getMaterial());
153 if (!mat)
154 continue;
155
156 double rho = mat->getDensity();
157 double cp = mat->getHeatCapacity();
158
159 // 注意：bcManager 施加的 Flux 是 Q (功率)，
160 // PD Kernel 计算出的也是源项率
161 // dT/dt = Net_Rate / (rho * Cp)
162 double heatRate = ratePtr[i];
163 double deltaT = (heatRate / (rho * cp)) * dt;
164
165 tempPtr[i] += deltaT;
166 }
167
168 // (F) 施加固定温度约束 (Dirichlet) - 时间积分后重新强制覆盖
169 bcManager.applyConstraints();
170 }
171
172 LOG_INFO("PD Solver Simulation Finished.");
173}

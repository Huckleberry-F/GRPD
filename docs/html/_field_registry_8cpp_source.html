23void FieldRegistry::registerFieldType(const std::string &type,
24 FieldCreatorFunc creator) {
25 if (registryMap_.find(type) != registryMap_.end()) {
26 LOG_WARNING("[FieldRegistry] Field type '" + type +
27 "' already registered, overwriting.");
28 } else {
29 LOG_INFO("[FieldRegistry] Field type '" + type + "' registered.");
30 }
31 registryMap_[type] = creator;
32}

34// ---------------------------------------------------------------------------
35// 通过工厂创建物理场实例
36// ---------------------------------------------------------------------------
37std::unique_ptr<Field> FieldRegistry::createField(const std::string &type,
38 const std::string &name) {
39 auto it = registryMap_.find(type);
40 if (it == registryMap_.end()) {
41 LOG_ERROR("[FieldRegistry] Create Field failed: unknown type '" + type +
42 "'!");
43 throw std::runtime_error("Unknown Field type: " + type);
44 }
45 return it->second(name);
46}

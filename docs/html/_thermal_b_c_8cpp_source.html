61void ConvectionBC::apply() {
62 double currentT = temperature_->get(particleId_);
63 double qConv = hConv_ * (tInf_ - currentT);
64 tempRate_->add(particleId_, qConv);
65 heatFlux_->add(particleId_, qConv);

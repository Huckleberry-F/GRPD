23// ---------------------------------------------------------------------------
24// trim: 去除字符串首尾空白字符（空格、制表符、回车、换行）
25// ---------------------------------------------------------------------------
26std::string GrpdReader::trim(const std::string &s) {
27 auto start = s.find_first_not_of(" \t\r\n");
28 if (start == std::string::npos)
29 return "";
30 auto end = s.find_last_not_of(" \t\r\n");
31 return s.substr(start, end - start + 1);
32}

34// ---------------------------------------------------------------------------
35// tokenizeCsv: 将逗号分隔的一行文本分割为修剪后的字段数组
36// ---------------------------------------------------------------------------
37std::vector<std::string> GrpdReader::tokenizeCsv(const std::string &line) {
38 std::vector<std::string> tokens;
39 std::istringstream ss(line);
40 std::string token;
41 while (std::getline(ss, token, ',')) {
42 tokens.push_back(trim(token));
43 }
44 return tokens;
45}

47// ---------------------------------------------------------------------------
48// scanFile: 通用文件扫描器
49// 封装：文件打开 → 逐行读取 → 空行/注释跳过 → 段标记状态切换
50// 遇到有效数据行时，调用 callback(state, tokens, lineNumber)
51// ---------------------------------------------------------------------------
52bool GrpdReader::scanFile(const std::string &filepath, LineCallback callback,
53 const std::string &logPrefix) {
54 // 1. 打开文件
55 std::ifstream file(filepath);
56 if (!file.is_open()) {
57 LOG_ERROR("[" + logPrefix + "] Error: cannot open file \"" + filepath +
58 "\"");
59 return false;
60 }
61
62 // 2. 状态机初始化
63 ParseState state = ParseState::IDLE;
64 std::string line;
65 int lineNumber = 0;
66
67 // 3. 逐行扫描
68 while (std::getline(file, line)) {
69 lineNumber++;
70
71 // 去除首尾空白
72 std::string trimmedLine = trim(line);
73
74 // 跳过空行
75 if (trimmedLine.empty()) {
76 continue;
77 }
78
79 // 跳过注释行（以 # 开头）
80 if (trimmedLine[0] == '#') {
81 continue;
82 }
83
84 // ----- 段标记检测：切换状态 -----
85 if (trimmedLine == "*PARTICLE") {
86 state = ParseState::READING_PARTICLES;
87 LOG_INFO("[" + logPrefix + "] Entering *PARTICLE section (Line " +
88 std::to_string(lineNumber) + ")");
89 continue;
90 }
91
92 if (trimmedLine == "*LOAD") {
93 state = ParseState::READING_LOADS;
94 LOG_INFO("[" + logPrefix + "] Entering *LOAD section (Line " +
95 std::to_string(lineNumber) + ")");
96 continue;
97 }
98
99 // IDLE 状态下忽略数据行
100 if (state == ParseState::IDLE) {
101 continue;
102 }
103
104 // ----- 分割字段并调用回调 -----
105 auto tokens = tokenizeCsv(line);
106 if (!callback(state, tokens, lineNumber)) {
107 break; // 回调返回 false 则终止扫描
108 }
109 }
110
111 file.close();
112 return true;
113}
114
115// ---------------------------------------------------------------------------
116// read: 读取 *PARTICLE 段，填充粒子几何数据到 ParticleManager
117// *LOAD 段仅统计行数，不做实际处理
118// ---------------------------------------------------------------------------
119bool GrpdReader::read(const std::string &filepath,
120 GRPD::Model::ParticleManager &manager) {
121 int particleCount = 0;
122 int loadCount = 0;
123
124 bool ok = scanFile(
125 filepath,
126 [&](ParseState state, const std::vector<std::string> &tokens,
127 int lineNumber) -> bool {
128 // ===== *PARTICLE 段：解析 ID, PartID, MatID, X, Y, Z, Volume =====
129 if (state == ParseState::READING_PARTICLES) {
130 if (tokens.size() != 7) {
131 LOG_WARNING(
132 "[GrpdReader] Line " + std::to_string(lineNumber) +
133 " particle field count=" + std::to_string(tokens.size()) +
134 ", expected 7, skipped");
135 return true; // 跳过当前行，继续扫描
136 }
137
138 try {
139 int id = std::stoi(tokens[0]);
140 int partId = std::stoi(tokens[1]);
141 int matId = std::stoi(tokens[2]);
142 double x = std::stod(tokens[3]);
143 double y = std::stod(tokens[4]);
144 double z = std::stod(tokens[5]);
145 double vol = std::stod(tokens[6]);
146
147 manager.addParticle(id, partId, matId, x, y, z, vol);
148 particleCount++;
149 } catch (const std::exception &e) {
150 LOG_ERROR("[GrpdReader] Line " + std::to_string(lineNumber) +
151 " parse failed: " + std::string(e.what()));
152 }
153 }
154
155 // ===== *LOAD 段：仅计数，实际处理由 readLoads() 负责 =====
156 if (state == ParseState::READING_LOADS) {
157 loadCount++;
158 }
159
160 return true; // 继续扫描
161 },
162 "GrpdReader");
163
164 if (!ok)
165 return false;
166
167 // 打印汇总统计
168 LOG_INFO("[GrpdReader] ============================================");
169 LOG_INFO("[GrpdReader] File reading complete!");
170 LOG_INFO("[GrpdReader] Total particles: " + std::to_string(particleCount));
171 LOG_INFO("[GrpdReader] Total load entries: " + std::to_string(loadCount));
172 LOG_INFO("[GrpdReader] ============================================");
173
174 return true;
175}

177// ---------------------------------------------------------------------------
178// readLoads: 读取 *LOAD 段，将载荷通过 FieldManager 施加到物理场
179// 仅处理 *LOAD 段，忽略 *PARTICLE 段数据
180// ---------------------------------------------------------------------------
181bool GrpdReader::readLoads(const std::string &filepath,
182 GRPD::Core::PDSimulater &simulater) {
183 auto &fieldManager = simulater.getFieldManager();
184 auto &bcManager = simulater.getBCManager();
185
186 // 检查温度场是否已注册
187 if (!fieldManager.hasField("Temperature")) {
188 LOG_WARNING("[GrpdReader] Temperature field not registered. Skipping loads.");
189 return false;
190 }
191
192 // 获取温度场指针（用于 Dirichlet 初始值设置）
193 auto *tempField = fieldManager.getFieldAs<double>("Temperature");
194
195 int tempCount = 0;
196 int fluxCount = 0;
197 int convCount = 0;
198 int otherCount = 0;
199
200 bool ok = scanFile(
201 filepath,
202 [&](ParseState state, const std::vector<std::string> &tokens,
203 int lineNumber) -> bool {
204 if (state != ParseState::READING_LOADS)
205 return true;
206 if (tokens.size() < 4)
207 return true;
208
209 try {
210 int id = std::stoi(tokens[0]);
211 int bcId = std::stoi(tokens[1]);
212 std::string type = tokens[2];
213 std::vector<double> values;
214 for (size_t i = 3; i < tokens.size(); i++) {
215 values.push_back(std::stod(tokens[i]));
216 }
217
218 // 使用 BcID 作为 BC 名称前缀，实现分类标记
219 std::string bcName =
220 type + "_BC" + std::to_string(bcId) + "_P" + std::to_string(id);
221
222 // 通过 BCRegistry 工厂创建 BC 实例（消除 if/else）
223 auto bc = GRPD::BC::BCRegistry::getInstance().createBC(type, bcName);
224 if (!bc) {
225 LOG_WARNING("[GrpdReader] Unknown BC type: " + type);
226 otherCount++;
227 } else {
228 bc->initialize(fieldManager, id, values);
229
230 // Dirichlet 约束需要同步设置初始温度值
231 if (bc->isConstraint()) {
232 tempField->set(id, values[0]);
233 tempCount++;
234 } else if (type == "FLUX") {
235 auto *heatFluxField = fieldManager.getFieldAs<double>("HeatFlux");
236 if (heatFluxField) {
237 heatFluxField->set(id, values[0]);
238 }
239 fluxCount++;
240 } else {
241 convCount++;
242 }
243
244 bcManager.addBC(std::move(bc));
245 }
246 } catch (const std::exception &e) {
247 LOG_ERROR("[GrpdReader::readLoads] Line " +
248 std::to_string(lineNumber) +
249 " parse failed: " + std::string(e.what()));
250 }
251
252 return true;
253 },
254 "GrpdReader::readLoads");
255
256 if (!ok)
257 return false;
258
259 LOG_INFO("[GrpdReader::readLoads] Load statistics:");
260 LOG_INFO(" - Temperature (Dirichlet): " + std::to_string(tempCount));
261 LOG_INFO(" - Heat Flux (Neumann): " + std::to_string(fluxCount));
262 LOG_INFO(" - Convection (Robin): " + std::to_string(convCount));
263 if (otherCount > 0)
264 LOG_WARNING(" - Unknown types: " + std::to_string(otherCount));
265
266 return true;
267}

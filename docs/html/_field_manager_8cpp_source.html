40// 工厂创建接口：通过 FieldRegistry 动态创建物理场
41// ---------------------------------------------------------------------------
42Field *FieldManager::createField(const std::string &typeName,
43 const std::string &fieldName) {
44 // 幂等：同名场已存在则直接返回
45 auto it = fields_.find(fieldName);
46 if (it != fields_.end()) {
47 LOG_WARNING("[FieldManager] Field '" + fieldName +
48 "' already exists. Returning existing instance.");
49 return it->second.get();
50 }
51
52 // 通过 FieldRegistry 工厂创建
53 auto field = FieldRegistry::getInstance().createField(typeName, fieldName);
54 if (!field) {
55 LOG_ERROR("[FieldManager] Failed to create field '" + fieldName +
56 "' with type '" + typeName + "'.");
57 return nullptr;
58 }
59
60 Field *ptr = field.get();
61 fields_[fieldName] = std::move(field);
62 LOG_INFO("[FieldManager] Created field '" + fieldName + "' (type: " +
63 typeName + ").");
64 return ptr;
65}

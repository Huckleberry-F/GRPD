17void SolverEngine::InitModelPD() {
18 auto &ctx = this->pdContext_;
19
20 // =================================================================
21 // Generate the initial mesh and save it to a .grpd file
22 // =================================================================
23 LOG_INFO("Model pre-processing...");
24
25 LOG_INFO("Calling Python pre-processor to generate .grpd mesh...");
26
27 std::string pythonCommand =
28 "python ../../Input/generate_model.py ../../Input/PD.yaml";
29
30 // Execute the command via std::system and capture the return value
31 int pyStatus = std::system(pythonCommand.c_str());
32
33 // Check if the Python script executed successfully (0 means success)
34 if (pyStatus != 0) {
35 LOG_ERROR("Python pre-processor failed! Please check if python is "
36 "installed and paths are correct.");
37
38 LOG_ERROR("Exit code: " + std::to_string(pyStatus));
39
40 exit(EXIT_FAILURE); // Terminate the program immediately
41 }
42
43 LOG_INFO("Mesh generated successfully by Python.");
44
45 // =================================================================
46 // Parse YAML to get OutputGrpd path and derive model name
47 // =================================================================
48 LOG_INFO("Parsing PD.yaml to extract model configuration...");
49 std::string yamlPath = "../../Input/PD.yaml";
50 std::string grpdPath = "";
51 std::string currentModelName = "";
52
53 try {
54 YAML::Node config = YAML::LoadFile(yamlPath);
55
56 // 1. 采用版本 B 的高防御力：精准检查节点是否存在
57 if (config["Assembly"] && config["Assembly"]["OutputGrpd"]) {
58 grpdPath = config["Assembly"]["OutputGrpd"].as<std::string>();
59
60 // 2. 采用版本 A 的极简优雅：使用 C++17 filesystem 提取文件名
61 std::filesystem::path p(grpdPath);
62 currentModelName = p.stem().string();
63
64 } else {
65 LOG_ERROR("Key [Assembly][OutputGrpd] not found in YAML file!");
66 exit(EXIT_FAILURE);
67 }
68 } catch (const YAML::Exception &e) {
69 LOG_ERROR("Failed to parse YAML file: " + std::string(e.what()));
70 exit(EXIT_FAILURE);
71 }
72
73 // =================================================================
74 // Define current model context (Replacing global registry with instance)
75 // =================================================================
76 ctx.setName(currentModelName);
77 LOG_INFO(std::string("[Context Manager] Formalized model name to: ") +
78 ctx.getName());
79 auto &manager = ctx.getParticleManager();
80
81 // =================================================================
82 // Read .grpd file and populate the specific particle manager
83 // =================================================================
84 LOG_INFO("Reading .grpd model file: " + grpdPath + " ...");
85
86 if (!GRPD::IO::GrpdReader::read(grpdPath, manager)) {
87 LOG_ERROR("Failed to read .grpd file!");
88 exit(EXIT_FAILURE);
89 }
90
91 // =================================================================
92 // Read done and print the total number of particles
93 // =================================================================
94 LOG_INFO("Total particles loaded for model " + currentModelName + ": " +
95 std::to_string(manager.getTotalParticles()));
96
97 // =================================================================
98 // Verification: print info for the first 5 particles
99 // =================================================================
100 size_t previewCount =
101 std::min(static_cast<size_t>(5), manager.getTotalParticles());
102 for (size_t i = 0; i < previewCount; i++) {
103 const auto &p = manager.getParticle(static_cast<int>(i));
104 LOG_INFO(" Particle[" + std::to_string(p.getId()) +
105 "] Part=" + std::to_string(p.getPartId()) + " Pos=(" +
106 std::to_string(p.getCoords()[0]) + ", " +
107 std::to_string(p.getCoords()[1]) + ", " +
108 std::to_string(p.getCoords()[2]) + ")" +
109 " Vol=" + std::to_string(p.getVolume()));
110 }
111}

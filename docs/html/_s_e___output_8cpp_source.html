18void SolverEngine::OutputPD() {
19 const auto &currentModel = this->pdContext_;
20 LOG_INFO("Starting data export process...");
21
22 // =================================================================
23 // 1. Parse YAML to find the current model context name
24 // =================================================================
25 std::string yamlPath = "../../Input/PD.yaml";
26 std::string grpdPath = "";
27 std::string currentModelName = currentModel.getName();
28 GRPD::IO::fileFormat format = GRPD::IO::ascii; // default to ascii
29 std::string customWriterName = "";
30 std::vector<std::pair<std::string, int>>
31 variablesToOutput; // {Name, Dimension}
32
33 try {
34 YAML::Node config = YAML::LoadFile(yamlPath);
35
36 if (config["Assembly"] && config["Assembly"]["OutputGrpd"]) {
37 // We already have the model name from the passed object
38 } else {
39 LOG_ERROR("Key [Assembly][OutputGrpd] not found in YAML file!");
40 return;
41 }
42
43 if (config["Writer"]) {
44 if (config["Writer"]["Name"]) {
45 customWriterName = config["Writer"]["Name"].as<std::string>();
46 }
47 if (config["Writer"]["Type"]) {
48 std::string typeStr = config["Writer"]["Type"].as<std::string>();
49 if (typeStr == "BINARY" || typeStr == "binary") {
50 format = GRPD::IO::binary;
51 } else if (typeStr == "ASCII" || typeStr == "ascii") {
52 format = GRPD::IO::ascii;
53 } else {
54 LOG_INFO("Unknown Writer Type: " + typeStr +
55 ". Using default ASCII.");
56 }
57 }
58 if (config["Writer"]["Variables"]) {
59 for (auto it : config["Writer"]["Variables"]) {
60 std::string varName = it["Name"].as<std::string>();
61 int varDim = it["Dimension"] ? it["Dimension"].as<int>() : 1;
62 variablesToOutput.push_back({varName, varDim});
63 }
64 } else {
65 variablesToOutput.push_back({"Coords", 3});
66 variablesToOutput.push_back({"PartID", 1});
67 variablesToOutput.push_back({"Volume", 1});
68 }
69 }
70 } catch (const YAML::Exception &e) {
71 LOG_ERROR("Failed to parse YAML file in Write(): " + std::string(e.what()));
72 return;
73 }
74
75 // =================================================================
76 // 2. Fetch the corresponding ParticleManager from model context
77 // =================================================================
78 LOG_INFO("Fetching model context for export: " + currentModelName);
79
80 try {
81 const auto &manager = currentModel.getParticleManager();
82 const auto &fieldManager = currentModel.getFieldManager();
83
84 // =================================================================
85 // 3. Export data using VtkWriter
86 // =================================================================
87 std::string finalOutputName =
88 customWriterName.empty() ? currentModelName : customWriterName;
89 std::string vtkOutputPath =
90 "../../Output/" + finalOutputName + "_output.vtk";
91 LOG_INFO("Exporting to VTK format: " + vtkOutputPath);
92
93 int numParticles = static_cast<int>(manager.getTotalParticles());
94
95 if (numParticles == 0) {
96 LOG_INFO("No particles to export for model: " + currentModelName);
97 return;
98 }
99
100 GRPD::IO::VtkWriter writer(vtkOutputPath, numParticles, format);
101
102 // Set points geometry
103 double *coordsPtr = manager.getGeomDataPtr(GRPD::Model::ModelVar::Coords);
104 if (coordsPtr) {
105 writer.setPointsInfo(GRPD::IO::Float64, coordsPtr, "Coordinates", 3);
106 } else {
107 LOG_ERROR("Failed to extract Coordinates. Aborting write.");
108 return;
109 }
110
111 // Set physical variables dynamically based on user config
112 for (const auto &[varName, yamlDim] : variablesToOutput) {
113 if (varName == "Coords") {
114 continue; // Coordinates are handled separately above
115 }
116
117 // ---------------------------------------------------------------
118 // 优先从 FieldManager 查询物理场变量
119 // FieldManager 统一存储所有物理场（Temperature, HeatFlux 等）
120 // ---------------------------------------------------------------
121 if (fieldManager.hasField(varName)) {
122 const auto *field = fieldManager.getField(varName);
123 if (field && field->rawPtr()) {
124 writer.setPointsVariable(GRPD::IO::Float64,
125 const_cast<double *>(field->rawPtr()),
126 varName.c_str(), yamlDim);
127 LOG_INFO("Exported field '" + varName + "' from FieldManager.");
128 } else {
129 LOG_WARNING("Field '" + varName + "' data pointer is null. Skipping.");
130 }
131 continue;
132 }
133
134 // ---------------------------------------------------------------
135 // 通用路由：来自 ParticleManager 的几何/拓扑变量
136 // ---------------------------------------------------------------
137 GRPD::Model::ModelVar enumName;
138 GRPD::Model::ParticleManager::VarDataType dataType;
139 int managerDim = 1;
140
141 if (GRPD::Model::ParticleManager::getModelVarInfo(varName, enumName,
142 dataType, managerDim)) {
143 void *dataPtr = nullptr;
144 GRPD::IO::dataType vtkType = GRPD::IO::Float64;
145
146 // Fetch the corresponding data pointer based on its specific type
147 if (dataType == GRPD::Model::ParticleManager::VarDataType::Double) {
148 dataPtr = manager.getGeomDataPtr(enumName);
149 vtkType = GRPD::IO::Float64;
150 } else if (dataType == GRPD::Model::ParticleManager::VarDataType::Int) {
151 dataPtr = manager.getIntDataPtr(enumName);
152 vtkType = GRPD::IO::Int32;
153 }
154
155 // Use the dimension specified in YAML configuration
156 int finalDim = yamlDim;
157
158 // If pointer extraction was successful, pass it to VTK Writer
159 if (dataPtr) {
160 writer.setPointsVariable(vtkType, dataPtr, varName.c_str(), finalDim);
161 } else {
162 LOG_WARNING("Data array for " + varName +
163 " is empty or null. Skipping.");
164 }
165 } else {
166 LOG_WARNING("Unknown output variable requested: " + varName +
167 ". Skipping.");
168 }
169 }
170 // =================================================================
171 // 4. Execute writing
172 // =================================================================
173 writer.write();
174
175 LOG_INFO("Successfully exported data for model: " + currentModelName);
176
177 } catch (const std::exception &e) {
178 LOG_ERROR("Write process aborted: " + std::string(e.what()));
179 }
180}

18void SolverEngine::InitConditionsPD() {
19 auto &ctx = this->pdContext_;
20 LOG_INFO("Entering Conditions Initialization Phase...");
21
22 // =================================================================
23 // 1. 解析 YAML 配置
24 // =================================================================
25 std::string yamlPath = "../../Input/PD.yaml";
26 YAML::Node config;
27 try {
28 config = YAML::LoadFile(yamlPath);
29 } catch (const YAML::BadFile &e) {
30 LOG_ERROR("Failed to load YAML file in InitConditionsPD: " +
31 std::string(e.what()));
32 exit(EXIT_FAILURE);
33 }
34
35 // =================================================================
36 // 2. 从 .grpd 文件的 *LOAD 段读取载荷并施加到物理场
37 // =================================================================
38 std::string grpdPath = "";
39 if (config["Assembly"] && config["Assembly"]["OutputGrpd"]) {
40 grpdPath = config["Assembly"]["OutputGrpd"].as<std::string>();
41 } else {
42 LOG_ERROR("[InitConditions] Key [Assembly][OutputGrpd] not found! "
43 "Cannot read load data.");
44 return;
45 }
46
47 LOG_INFO("[InitConditions] Reading load data from .grpd file: " + grpdPath);
48
49 if (!GRPD::IO::GrpdReader::readLoads(grpdPath, ctx)) {
50 LOG_ERROR("[InitConditions] Failed to read loads from .grpd file!");
51 return;
52 }
53
54 LOG_INFO("[InitConditions] Boundary conditions initialization complete.");
55}

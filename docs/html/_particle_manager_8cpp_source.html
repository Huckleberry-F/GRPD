17// addParticle: Emplace-construct into the global array
18// Safety check: Ensures the incoming ID matches the current array size
19// (enforcing the ID==index invariant)
20// ---------------------------------------------------------------------------
21void ParticleManager::addParticle(int id, int partId, int matId, double x,
22 double y, double z, double volume) {
23 // Assert ID continuity: incoming id must exactly equal current array length
24 if (static_cast<size_t>(id) != globalParticles_.size()) {
25 LOG_ERROR("[ParticleManager] Error: particle ID=" + std::to_string(id) +
26 " does not match expected index " +
27 std::to_string(globalParticles_.size()) +
28 "! Violates zero-based ID invariant!");
29 throw std::runtime_error(
30 "Particle ID does not match expected vector index.");
31 }
32
33 globalParticles_.emplace_back(id, partId, matId, x, y, z, volume);
34}

106bool ParticleManager::getModelVarInfo(const std::string &varName,
107 ModelVar &outVar, VarDataType &outType,
108 int &outDim) {
109 // Map structure: { "StringName", { ModelVarEnum, VarDataTypeEnum,
110 // DimensionValue } }
111 struct VarInfo {
112 ModelVar var;
113 VarDataType type;
114 int dim;
115 };
116
117 static const std::unordered_map<std::string, VarInfo> varMap = {
118 {"Coords", {ModelVar::Coords, VarDataType::Double, 3}},
119 {"Volume", {ModelVar::Volume, VarDataType::Double, 1}},
120
121 {"ID", {ModelVar::ID, VarDataType::Int, 1}},
122 {"PartID", {ModelVar::PartID, VarDataType::Int, 1}},
123 {"MatID", {ModelVar::MatID, VarDataType::Int, 1}}};
124
125 auto it = varMap.find(varName);
126 if (it != varMap.end()) {
127 outVar = it->second.var;
128 outType = it->second.type;
129 outDim = it->second.dim;
130 return true;
131 }
132 return false;
133}
